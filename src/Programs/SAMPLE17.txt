-TSO ALLOC F(PASS) DA('TEST.CITY') SHR REUSE
-TSO ALLOC F(LOG) DA('TEST.LOG') SHR REUSE
-WRITE PASS &CITY
TABLE FILE SALES
HEADING CENTER
"LOWEST MONTHLY SALES FOR &CITY"
" "
PRINT DATE PROD_CODE
BY LOWEST 1 UNIT_SOLD
BY STORE_CODE
BY CITY
WHERE CITY EQ '&CITY'
FOOTING CENTER
"CALCULATED AS OF &DATE"
ON TABLE SAVE AS INFO
END
-RUN
-INCLUDE SAMPLE15
MODIFY FILE SALES
COMPUTE
TODAY/I6=&YMD;
CITY='&CITY';
FIXFORM X5 STORE_CODE/A3 X15 DATE/A4 PROD_CODE/A3
MATCH STORE_CODE DATE PROD_CODE
ON MATCH TYPE ON LOG
"<STORE_CODE><DATE><PROD_CODE><TODAY>"
ON MATCH DELETE
ON NOMATCH REJECT
DATA ON INFO
END
-RUN
EX SLRPT3
-RUN
-GOTO TOP
-QUIT
