DEFINE FILE EMPDATA
CURR_SAL/D12.2M = CSAL;
FIRST_NAME/A10 = FN;
EID/A9 = PIN;
END
-*Start MATCH.
MATCH FILE EMPLOYEE
SUM CURR_SAL AS 'CURRENT'
FIRST_NAME AS 'FIRST'
BY EID AS 'SSN'
-*Concatenate file EMPDATA to EMPLOYEE to form first MATCH answer set.
MORE
FILE EMPDATA
RUN
-*Second MATCH answer set:
FILE TRAINING
PRINT EXPENSES
BY PIN AS 'SSN'
AFTER MATCH HOLD OLD-OR-NEW
END
-*Print merged file:
TABLE FILE HOLD
PRINT *
END
