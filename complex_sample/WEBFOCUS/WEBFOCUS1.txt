<script>
  function validateCity() {
    const cityInput = document.getElementById("city_input");
    const cityValue = cityInput.value.trim();

    if (cityValue.length !== 3) {
      alert("City must be 3 characters long.");
      return false;
    }

    if (!cityValue.startsWith('I')) {
      alert("City must start with the letter 'I'.");
      return false;
    }

    return true;
  }
</script>
